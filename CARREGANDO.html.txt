    <div id="loading-text">Abrindo sua conta</div>

    <script>
        const texts = [
            "Abrindo sua conta",
            "Carregando Séries e Filmes",
            "Carregando design",
            "Redirecionando"
        ];
        let textIndex = 0;
        let dotCount = 0;
        const textElement = document.getElementById('loading-text');

        function updateText() {
            textElement.innerText = texts[textIndex] + '.'.repeat(dotCount);
        }

        const textInterval = setInterval(() => {
            dotCount = (dotCount + 1) % 4;
            updateText();
        }, 400);

        const changeTextInterval = setInterval(() => {
            if (textIndex < texts.length - 1) {
                textIndex++;
            } else {
                clearInterval(changeTextInterval);
                setTimeout(() => {
                    clearInterval(textInterval);
                }, 400); // Aguarda a última atualização de texto
            }
        }, 4000);

        // Atualiza o texto inicial
        updateText();
    </script>
